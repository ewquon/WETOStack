def create_field_name(field_string) -> str:
    parts = field_string.split('_')
    capitalized = [part.capitalize() for part in parts]
    name = ' '.join(capitalized)
    return name

def create_data_table(fields: list[str]) -> DataTable:
    columns = [
        TableColumn(field=field, title=create_field_name(field))
        for field in fields
    ]
    data_table = DataTable(
        index_position=None,
        source=source,
        columns=columns,
        # width=400,
        # height=280,
    )
    return data_table

Attribute Descriptions#

We need a way to relate software projects to each other to understand how the combination of software projects defines the landscape of WETO funded software. These attributes are intended to answer questions about the health, quality, and maturity of the portfolio.
